% 正文内的居中标题格式，用于作者简历等。（字号为小3号）
\newcommand{\chaptercenter}[1]
{
    \phantomsection
    \addcontentsline{toc}{chapter}{#1}
    \markboth{#1}{#1}
    \begin{center}
        \vspace*{-6pt}
        \fangsong\zihao{-3}\bfseries{#1}
        \vspace*{6pt}
    \end{center}
    \hspace*{1.5em}
}

% previous page中的居中标题格式，用于致谢、摘要等。（字号为3号）
\newcommand{\chapterprev}[1]
{
    \phantomsection
    \addcontentsline{toc}{chapter}{#1}
    \markboth{#1}{#1}
    \begin{center}
        \vspace*{-6pt}
        \fangsong\zihao{3}\bfseries{#1}
        \vspace*{6pt}
    \end{center}
    \hspace*{1.5em}
}
